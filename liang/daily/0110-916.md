class Solution {
public:
    bool check_if_in_arr(vector<int>& arr1, vector<int>& arr2){
        for (int i=0; i<arr1.size();++i){
            if (arr1[i]>0){
                if (arr2[i]<arr1[i])
                    return false;
            }
        }
        return true;
    }
    vector<string> wordSubsets(vector<string>& words1, vector<string>& words2) {
        
        int m = words1.size();
        int n = words2.size();
        int idx;

        vector<int> arr(26,0);
        for (int i=0;i<n;++i){
            vector<int> record(26,0);
            for (int j=0;j<words2[i].size();++j){
                idx = words2[i][j]-'a';
                record[idx]++;
                
                // give an 1 (or the max)
                arr[idx] = max(arr[idx],record[idx]);
            }
        }

        // iterate through words1 string
        vector<string> ans;
        
        for (int i=0; i<m; ++i){
            vector<int> cmp_arr(26,0);
            for (int j=0; j<words1[i].size(); ++j){
                idx = words1[i][j] - 'a';
                cmp_arr[idx]++;
            }
            // check if it is zero
            if (check_if_in_arr(arr,cmp_arr))
                ans.push_back(words1[i]);
        }
        return ans;

    }
};

                count[idx]++;

                w2_count[idx] = max(w2_count[idx],count[idx]);
            }
        }

        // iterate
        vector<string> ans;
        for (int i=0;i<m;++i){
            int w1_arr[26]={0};

            for (int j=0;j<words1[i][j];++j){

            // compare
            if (is_Universal(w2_count, w1_arr))

        return ans;

                    return false;

class Solution {
public:
    vector<string> wordSubsets(vector<string>& words1, vector<string>& words2) {
        int m = words1.size();
        int n = words2.size();

        array<int,26> w2_count = {0};
        int idx;
        for (int i=0;i<n;++i){
            int count[26] = {0};
            for (auto w : words2[i]){
                idx = w -'a';
                count[idx]++;
                w2_count[idx] = max(w2_count[idx],count[idx]);
            }
        }

        // iterate
        vector<string> ans;
        for (int i=0;i<m;++i){
            array<int,26> w1_arr = {0};
            // w1 - voc table
            for (auto w : words1[i])
            {
                w1_arr[w-'a']++;
            }
            // compare
            if (is_Universal(w2_count, w1_arr))
                ans.push_back(words1[i]);

        }
        return ans;


    }
private:
    
    // compare 2 arrays
    bool is_Universal(const array<int,26> arr1, const array<int,26>& arr2){
        for (int i=0;i<26;++i){
            if (arr2[i]<arr1[i]){
                return false;
            }
        }
        return true;
    }
    
};
